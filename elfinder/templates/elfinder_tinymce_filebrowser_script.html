function elFinderBrowser (field_name, url, type, win) { var elfinder_url = '{% url "elfinder_tinymce_filebrowser_dialog" %}'; tinyMCE.activeEditor.windowManager.open({ file: elfinder_url, title: 'elFinder 2.0', width: 1400, height: 450, resizable: true, popup_css: false, // Disable TinyMCE's default popup CSS close_previous: false }, { setUrl: function (url) { win.document.getElementById(field_name).value = url; } }); return false; }
